47float Neuron::activate(const Eigen::VectorXf &inputs)
48{
49 // Calculate the weighted sum of the inputs
50 e_lastInput = inputs;

52
53 // Calculate the weighted sum of the inputs

67void Neuron::update()
68{
69 // Update the weights and bias
70 float error = _learningRate * _delta;
71 for (std::size_t i = 0; i < _nSizeWeights; i++)
72 {
73 _weights[i] -= e_lastInput[i] * error;
74 }
75
76 // Update the bias
