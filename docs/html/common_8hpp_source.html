173template <typename T>
174void normalizeVector(std::vector<T> &vec)
175{
176 T maxElement = *std::max_element(vec.begin(), vec.end());
177 T minElement = *std::min_element(vec.begin(), vec.end());
178 T minMax = maxElement - minElement;
179
180 // https://www.statology.org/normalize-data-between-0-and-1/
181 for (std::size_t i = 0; i < vec.size(); i++)
182 {
183 T xi = vec[i];
184 vec.at(i) = (xi - minElement) / minMax;
185 }
186}

237TrainTestSplit<T> createTrainTestSplit(
238 const std::vector<std::vector<T>> &features,
239 const std::vector<std::vector<T>> &targets,
240 float splitRatio)
241{
242 TrainTestSplit<T> tts;
243 // Reserve the memory for the vectors
244 tts.trainFeatures.reserve(features.size() * splitRatio);
245 tts.testFeatures.reserve(features.size() * (1 - splitRatio));
246 tts.trainTargets.reserve(targets.size() * splitRatio);
247 tts.testTargets.reserve(targets.size() * (1 - splitRatio));
248
249 std::random_device rd;
250 std::mt19937 gen(rd());
251 std::uniform_int_distribution<> dis(0, 1);
252
253 for (std::size_t i = 0; i < features.size(); i++)
254 {
255 if (i < features.size() * splitRatio)
256 {
257 tts.trainFeatures.emplace_back(features[i]);
258 tts.trainTargets.emplace_back(targets[i]);
259 }
260 else
261 {
262 tts.testFeatures.emplace_back(features[i]);
263 tts.testTargets.emplace_back(targets[i]);
264 }
265 }
266 return tts;
