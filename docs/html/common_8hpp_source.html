159template<typename T>
160void normalizeVector(std::vector<T> &vec)
161{
162 T maxElement = *std::max_element(vec.begin(), vec.end());
163 T minElement = *std::min_element(vec.begin(), vec.end());
164 T minMax = maxElement - minElement;
165
166 // https://www.statology.org/normalize-data-between-0-and-1/
167 for (std::size_t i = 0; i < vec.size(); i++) {
168 T xi = vec[i];
169 vec.at(i) = (xi - minElement) / minMax;
170 }
171}

173template<typename T>
174TrainTestSplit<T> createTrainTestSplit(
175 const std::vector<std::vector<T>>& features,
176 const std::vector<std::vector<T>>& targets,
177 float splitRatio)
178{
179 TrainTestSplit<T> tts;
180 // Reserve the memory for the vectors
181 tts.trainFeatures.reserve(features.size() * splitRatio);
182 tts.testFeatures.reserve(features.size() * (1 - splitRatio));
183 tts.trainTargets.reserve(targets.size() * splitRatio);
184 tts.testTargets.reserve(targets.size() * (1 - splitRatio));
185
186 for (std::size_t i = 0; i < features.size(); i++)
187 {
188 if (i < features.size() * splitRatio) {
189 tts.trainFeatures.emplace_back(features[i]);
190 tts.trainTargets.emplace_back(targets[i]);
191 } else {
192 tts.testFeatures.emplace_back(features[i]);
193 tts.testTargets.emplace_back(targets[i]);
194 }
195 }
196 return tts;
197}
