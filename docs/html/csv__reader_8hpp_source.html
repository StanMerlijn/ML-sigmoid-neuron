31std::vector<std::vector<std::string>> readCsv(const std::string &filename, char delimiter = ',')
32{
33 // Create a vector to store the rows
34 std::vector<std::vector<std::string>> rows;
35 std::ifstream file(filename);
36
37 // Check if the file is open
38 if (!file.is_open())
39 {
40 std::cerr << "Error: Could not open file " << filename << std::endl;
41 std::cerr << "From the current directory: " << std::filesystem::current_path() << std::endl;
42 return rows;
43 }
44
45 // Read the file line by line
46 std::string line;
47 while (std::getline(file, line))
48 {
49 std::stringstream ss(line);
50 std::vector<std::string> cols;
51 std::string col;
52
53 while (std::getline(ss, col, delimiter))
54 {
55 cols.push_back(col);
56 }
57
58 // Add the columns to the rows
59 rows.push_back(cols);
60 }
61
62 // Close the file
63 file.close();
64
65 return rows;
66}

244template <typename T>
245std::vector<T> maskData(std::vector<T> &data, std::vector<T> &mask)
246{
247 T off, on;
248 if constexpr (std::is_floating_point_v<T>)
249 {
250 off = 0.0f;
251 on = 1.0f;
252 }
253 else if constexpr (std::is_integral_v<T>)
254 {
255 off = 0;
256 on = 1;
257 }
258 else
259 {
260 throw std::runtime_error("Data type not supported");
261 }
262
263 std::vector<T> maskedData;
264 maskedData.reserve(data.size() * mask.size());
265 for (std::size_t i = 0; i < data.size(); i++)
266 {
267 for (std::size_t j = 0; j < mask.size(); j++)
268 {
269 if (data.at(i) == mask.at(j))
270 {
271 maskedData.emplace_back(on);
272 }
273 else
274 {
275 maskedData.emplace_back(off);
276 }
277 }
278 }
279 return maskedData;
