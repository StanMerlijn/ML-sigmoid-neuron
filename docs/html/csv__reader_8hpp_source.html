32std::vector<std::vector<std::string>> read_csv(const std::string& filename, char delimiter=',')
33{
34 // Create a vector to store the rows
35 std::vector<std::vector<std::string>> rows;
36 std::ifstream file(filename);
37
38 // Check if the file is open
39 if (!file.is_open()) {
40 std::cerr << "Error: Could not open file " << filename << std::endl;
41 return rows;
42 }
43
44 // Read the file line by line
45 std::string line;
46 while (std::getline(file, line)) {
47 std::stringstream ss(line);
48 std::vector<std::string> cols;
49 std::string col;
50
51 while (std::getline(ss, col, delimiter)) {
52 cols.push_back(col);
53 }
54
55 // Add the columns to the rows
56 rows.push_back(cols);
57 }
58
59 // Close the file
60 file.close();
61
62 return rows;
63}

230template<typename T>
231std::vector<T> maskData(std::vector<T>& data, std::vector<T>& mask)
232{
233 T off, on;
234 if constexpr (std::is_floating_point_v<T>) {
235 off = 0.1f;
236 on = 1.0f;
237 } else if constexpr (std::is_integral_v<T>) {
238 off = 0;
239 on = 1;
240 } else {
241 throw std::runtime_error("Data type not supported");
242 }
243
244 std::vector<T> maskedData;
245 maskedData.reserve(data.size() * mask.size());
246 for (std::size_t i = 0; i < data.size(); i++)
247 {
248 for (std::size_t j = 0; j < mask.size();j++)
249 {
250 if (data.at(i) == mask.at(j)) {
251 maskedData.emplace_back(on);
252 } else {
253 maskedData.emplace_back(off);
254 }
255 }
256 }
257 return maskedData;
